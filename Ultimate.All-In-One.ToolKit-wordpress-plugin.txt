1. قم بإنشاء مجلد على جهازك باسم ultimate-all-in-one-plugin.
2. بداخل المجلد، أنشئ ملفاً باسم ultimate-all-in-one-plugin.php وضع فيه الكود البرمجي الطويل أدناه بالكامل.

function uaio_get_facebook_posts() {
   $page_id      = get_option('uaio_fb_page_id'); 
   $access_token = get_option('uaio_fb_token');
   
   if ( empty($page_id) || empty($access_token) ) return false;

   $transient_key = 'uaio_fb_posts_' . $page_id;
   $cached_posts  = get_transient( $transient_key );

   if ( false !== $cached_posts ) return $cached_posts;

   $url = "https://graph.facebook.com/v19.0/{$page_id}/feed?fields=message,full_picture,permalink_url,created_time&limit=5&access_token={$access_token}";
   $response = wp_remote_get( $url );

   if ( is_wp_error( $response ) ) return false;

   $body = wp_remote_retrieve_body( $response );
   $data = json_decode( $body, true );

   if ( isset( $data['data'] ) ) {
       set_transient( $transient_key, $data['data'], 3600 ); // كاش ساعة
       return $data['data'];
   }
   return false;
}

add_shortcode( 'uaio_facebook_feed', 'uaio_facebook_feed_shortcode' );
function uaio_facebook_feed_shortcode() {
   $posts = uaio_get_facebook_posts();
   if ( ! $posts ) return '<p>يرجى ضبط إعدادات فيسبوك الحية من لوحة التحكم لعرض المنشورات.</p>';

   $output = '<div class="uaio-fb-container" style="max-width:600px; margin:0 auto; direction:rtl; text-align:right;">';
   foreach ( $posts as $post ) {
       $message = isset( $post['message'] ) ? esc_html( $post['message'] ) : '';
       $image   = isset( $post['full_picture'] ) ? esc_url( $post['full_picture'] ) : '';
       $link    = isset( $post['permalink_url'] ) ? esc_url( $post['permalink_url'] ) : '#';
       $date    = isset( $post['created_time'] ) ? date( 'Y-m-d', strtotime( $post['created_time'] ) ) : '';

       if ( empty($message) && empty($image) ) continue;

       $output .= '<div style="border:1px solid #ddd; padding:15px; margin-bottom:20px; border-radius:8px; background:#fff;">';
       if ( $message ) $output .= '<p style="font-size:15px; line-height:1.6;">' . nl2br($message) . '</p>';
       if ( $image ) $output .= '<img src="'.$image.'" style="max-width:100%; height:auto; border-radius:4px; margin-bottom:10px;" />';
       $output .= '<div style="font-size:12px; color:#777;">التاريخ: '.$date.' | <a href="'.$link.'" target="_blank" style="color:#1877f2;">عرض على فيسبوك</a></div>';
       $output .= '</div>';
   }
   $output .= '</div>';
   return $output;
}

add_shortcode( 'uaio_embed_video', 'uaio_video_embedder_shortcode' );
function uaio_video_embedder_shortcode( $atts ) {
   $atts = shortcode_atts( array( 'url' => '' ), $atts );
   $url = esc_url( $atts['url'] );
   if ( empty( $url ) ) return '';

   $output = '<div class="uaio-video-wrapper" style="position:relative; padding-bottom:56.25%; height:0; overflow:hidden; max-width:100%; margin:20px auto; background:#000; border-radius:8px;">';
   
   if ( preg_match( '/(youtube\.com|youtu\.be)/', $url ) ) {
       if ( preg_match( '/(v=|\/embed\/|youtu\.be\/)([\w-]+)/', $url, $matches ) ) {
           $output .= '<iframe src="https://www.youtube.com/embed/' . esc_attr( $matches[2] ) . '" style="position:absolute; top:0; left:0; width:100%; height:100%; border:0;" allowfullscreen></iframe>';
       }
   } elseif ( preg_match( '/facebook\.com/', $url ) ) {
       $output .= '<iframe src="https://www.facebook.com/plugins/video.php?href=' . urlencode( $url ) . '&show_text=0" style="position:absolute; top:0; left:0; width:100%; height:100%; border:0;" allowfullscreen></iframe>';
   } elseif ( preg_match( '/instagram\.com/', $url ) ) {
       $output .= '<iframe src="' . rtrim( $url, '/' ) . '/embed/" style="position:absolute; top:0; left:0; width:100%; height:100%; border:0;" scrolling="no"></iframe>';
   } elseif ( preg_match( '/tiktok\.com/', $url ) && preg_match( '/video\/(\d+)/', $url, $matches ) ) {
       $output .= '<iframe src="https://www.tiktok.com/embed/v2/' . esc_attr( $matches[1] ) . '" style="position:absolute; top:0; left:0; width:100%; height:100%; border:0;" allowfullscreen></iframe>';
   } else {
       $output .= wp_oembed_get( $url );
   }

   $output .= '</div>';
   return $output;
}

add_shortcode( 'uaio_contact_form', 'uaio_contact_form_shortcode' );
function uaio_contact_form_shortcode() {
   $msg_status = '';
   if ( isset( $_POST['uaio_send_contact'] ) && wp_verify_nonce( $_POST['uaio_nonce'], 'uaio_contact_act' ) ) {
       $name    = sanitize_text_field( $_POST['uaio_name'] );
       $email   = sanitize_email( $_POST['uaio_email'] );
       $message = esc_textarea( $_POST['uaio_message'] );

       if ( empty($name) || !is_email($email) || empty($message) ) {
           $msg_status = '<div style="background:#f8d7da; color:#721c24; padding:10px; margin-bottom:15px; border-radius:4px;">يرجى التحقق من المدخلات بشكل صحيح.</div>';
       } else {
           $to      = get_option( 'admin_email' );
           $body    = "رسالة جديدة من: $name \nبريد إلكتروني: $email \n\nنص الرسالة:\n$message";
           $headers = array('Content-Type: text/plain; charset=UTF-8', 'Reply-To: ' . $email);
           
           if ( wp_mail( $to, "رسالة تواصل جديدة", $body, $headers ) ) {
               $msg_status = '<div style="background:#d4edda; color:#155724; padding:10px; margin-bottom:15px; border-radius:4px;">تم الإرسال بنجاح! شكراً لك.</div>';
           }
       }
   }

   ob_start();
   ?>
   <form method="POST" style="max-width:500px; margin:20px auto; padding:20px; border:1px solid #eee; border-radius:8px; background:#fafafa; direction:rtl; text-align:right;">
       <?php echo $msg_status; ?>
       <?php wp_nonce_field( 'uaio_contact_act', 'uaio_nonce' ); ?>
       <p><label>الاسم الكامل *</label><br/><input type="text" name="uaio_name" required style="width:100%; padding:8px; border:1px solid #ccc; border-radius:4px;" /></p>
       <p><label>البريد الإلكتروني *</label><br/><input type="email" name="uaio_email" required style="width:100%; padding:8px; border:1px solid #ccc; border-radius:4px;" /></p>
       <p><label>الرسالة *</label><br/><textarea name="uaio_message" rows="5" required style="width:100%; padding:8px; border:1px solid #ccc; border-radius:4px;"></textarea></p>
       <p><input type="submit" name="uaio_send_contact" value="إرسال الآن" style="background:#0073aa; color:#fff; border:none; padding:10px 20px; border-radius:4px; cursor:pointer; width:100%; font-weight:bold;" /></p>
   </form>
   <?php
   return ob_get_clean();
}

add_shortcode( 'uaio_gallery', 'uaio_gallery_shortcode' );
function uaio_gallery_shortcode( $atts ) {
   $atts = shortcode_atts( array( 'ids' => '' ), $atts );
   if ( empty( $atts['ids'] ) ) return '';

   $image_ids = explode( ',', $atts['ids'] );
   $output = '<div class="uaio-gallery-grid" style="display:grid; grid-template-columns:repeat(auto-fill, minmax(150px,1fr)); gap:10px; margin:20px 0;">';

   foreach ( $image_ids as $id ) {
       $thumb = wp_get_attachment_image_src( trim($id), 'medium' );
       $full  = wp_get_attachment_image_src( trim($id), 'large' );
       if ( $thumb && $full ) {
           $output .= '<div style="height:120px; border-radius:4px; overflow:hidden;">';
           $output .= '<a href="'.esc_url($full[0]).'" class="uaio-lightbox-open"><img src="'.esc_url($thumb[0]).'" style="width:100%; height:100%; object-fit:cover; display:block; cursor:pointer;" /></a>';
           $output .= '</div>';
       }
   }
   $output .= '</div>';
   return $output;
}

// دالة خلفية PHP لمعالجة طلب الـ AI الآتي من الزائر عبر الـ API
add_action( 'wp_ajax_uaio_bot_response', 'uaio_bot_response_handler' );
add_action( 'wp_ajax_nopriv_uaio_bot_response', 'uaio_bot_response_handler' );
function uaio_bot_response_handler() {
   $msg = isset($_POST['msg']) ? sanitize_text_field($_POST['msg']) : '';
   $key = get_option('uaio_openai_key');

   if( empty($msg) || empty($key) ) {
       wp_send_json_error('مفتاح الـ API غير مضبوط أو الرسالة فارغة.');
   }

   $response = wp_remote_post( 'https://api.openai.com/v1/chat/completions', array(
       'headers' => array( 'Authorization' => 'Bearer ' . $key, 'Content-Type' => 'application/json' ),
       'body'    => json_encode(array(
           'model' => 'gpt-4o-mini',
           'messages' => array(
               array('role' => 'system', 'content' => 'أنت مساعد ذكي ومؤدب تجيب باختصار شديد باللغة العربية للرد على العملاء بموقع ووردبريس.'),
               array('role' => 'user', 'content' => $msg)
           ),
           'max_tokens' => 150
       )),
       'timeout' => 12
   ));

   if ( is_wp_error( $response ) ) wp_send_json_error();

   $res_body = json_decode( wp_remote_retrieve_body( $response ), true );
   if ( isset($res_body['choices'][0]['message']['content']) ) {
       wp_send_json_success( trim($res_body['choices'][0]['message']['content']) );
   } else {
       wp_send_json_error();
   }
}
